Scenario Description

Execute Python code in a secure AWS sandbox. Perfect for data analysis, calculations, visualizations, and more!

AgentCore SDK Usage (Application Side)
Reference
üìå Important: This code runs in YOUR application with AWS credentials from your .env file. It creates a Code Interpreter session and sends user code to the AWS AgentCore sandbox for execution. The sandbox code (shown below) runs in an isolated environment WITHOUT credentials.
import boto3 # Code to execute in the AgentCore sandbox code_to_execute = """ print("Hello from AgentCore!") """ # Initialize the bedrock-agentcore client (requires AWS credentials) client = boto3.client( "bedrock-agentcore", region_name="us-west-2", endpoint_url="https://bedrock-agentcore.us-west-2.amazonaws.com" ) # Start a Code Interpreter session session_response = client.start_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", name="my-code-session", sessionTimeoutSeconds=900 ) session_id = session_response["sessionId"] print(f"Started session: {session_id}") try: # Execute code in the session execute_response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="executeCode", arguments={ "language": "python", "code": code_to_execute } ) # Extract and print the text output from the stream for event in execute_response['stream']: if 'result' in event: result = event['result'] if 'content' in result: for content_item in result['content']: if content_item['type'] == 'text': print(content_item['text']) finally: # Stop the session when done client.stop_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id ) print(f"Stopped session: {session_id}")

Reference

import boto3 import json # Sample data files to create in the sandbox data_file_content = """Date,Sales,Region 2024-01-01,1500,North 2024-01-02,1800,South 2024-01-03,1200,East""" stats_script_content = """import pandas as pd df = pd.read_csv('data.csv') print('Sales Summary:') print(df.describe())""" files_to_create = [ { "path": "data.csv", "text": data_file_content }, { "path": "stats.py", "text": stats_script_content } ] # Initialize the bedrock-agentcore client client = boto3.client( "bedrock-agentcore", region_name="us-west-2", endpoint_url="https://bedrock-agentcore.us-west-2.amazonaws.com" ) # Start a Code Interpreter session session_response = client.start_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", name="file-mgmt-session", sessionTimeoutSeconds=900 ) session_id = session_response["sessionId"] try: # 1. Write files to the sandbox write_response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="writeFiles", arguments={"content": files_to_create} ) print("‚úÖ Files written successfully") # 2. List files in the sandbox list_response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="listFiles", arguments={"path": ""} ) for event in list_response['stream']: if 'result' in event: result = event['result'] if 'content' in result: for item in result['content']: if item['type'] == 'text': print(item['text']) # 3. Delete a file delete_response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="removeFiles", arguments={"paths": ["stats.py"]} ) print("üóëÔ∏è File deleted successfully") finally: # Stop the session client.stop_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id )
Scenario Description

Execute shell commands in the AWS sandbox using AgentCore SDK. Collect system information, run automation scripts, and interact with the Linux environment!

AgentCore SDK Usage (Application Side)
Reference
üìå Important: This code runs in YOUR application with AWS credentials from your .env file. It demonstrates how to execute shell commands in the AgentCore sandbox using the executeCommand operation.
import boto3 # Initialize the bedrock-agentcore client client = boto3.client( "bedrock-agentcore", region_name="us-west-2", endpoint_url="https://bedrock-agentcore.us-west-2.amazonaws.com" ) # Start a Code Interpreter session session_response = client.start_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", name="shell-command-session", sessionTimeoutSeconds=900 ) session_id = session_response["sessionId"] try: # 1. Get system information response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="executeCommand", arguments={"command": "uname -a"} ) for event in response['stream']: if 'result' in event: result = event['result'] if 'content' in result: for item in result['content']: if item['type'] == 'text': print("System:", item['text']) # 2. Check CPU information response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="executeCommand", arguments={"command": "nproc"} ) print("‚úÖ CPU cores checked") # 3. Check memory usage response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="executeCommand", arguments={"command": "free -h"} ) print("‚úÖ Memory info collected") # 4. Check disk usage response = client.invoke_code_interpreter( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id, name="executeCommand", arguments={"command": "df -h"} ) print("‚úÖ Disk usage analyzed") finally: # Stop the session client.stop_code_interpreter_session( codeInterpreterIdentifier="aws.codeinterpreter.v1", sessionId=session_id )
